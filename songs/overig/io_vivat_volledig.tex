\songtitle{Io Vivat}
\begin{verse*}
Io vivat! Io vivat,
Nostrorum sanitas!
Hoc est amoris poculum,
Doloris est antidotum!
Io vivat! Io vivat,
Nostrorum sanitas!
\end{verse*}
