\songtitle{Bruderlein Trink}
\begin{verse*}
Trink, trink, Brüderlein trink,
Laß doch die Sorgen zu Haus!
Trink, trink, Brüderlein trink,
Zieh doch die Stirn nicht so kraus,
\end{verse*}
