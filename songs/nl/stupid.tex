\songtitle{Stupid}
\begin{verse*}
Het is weer zover
Kriebels in m'n maag
wat doe ik ermee, stel ik hem de vraag
ik zie hem al lopen en draai me om
ik weet wel zeker, hij vind me vast dom...
\end{verse*}

\end{chorus}
